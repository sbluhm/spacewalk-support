202 int filter_packet(router_t r, nad_t nad) {
203  acl_t acl;
204  int ato, afrom, error = 0;
205  char *cur, *to = NULL, *from = NULL;
206 
207  ato = nad_find_attr(nad, 1, -1, "to", NULL);
208  afrom = nad_find_attr(nad, 1, -1, "from", NULL);
209  if(ato >= 0 && NAD_AVAL_L(nad,ato) > 0) {
210  to = (char *) malloc(sizeof(char) * (NAD_AVAL_L(nad, ato) + 1));
211  sprintf(to, "%.*s", NAD_AVAL_L(nad, ato), NAD_AVAL(nad, ato));
212  cur = strstr(to, "@"); /* skip node part */
213  if(cur != NULL)
214  cur = strstr(cur, "/");
215  else
216  cur = strstr(to, "/");
217  if(cur != NULL) *cur = '\0'; /* remove the resource part */
218  }
219  if(afrom >= 0 && NAD_AVAL_L(nad,afrom) > 0) {
220  from = (char *) malloc(sizeof(char) * (NAD_AVAL_L(nad, afrom) + 1));
221  sprintf(from, "%.*s", NAD_AVAL_L(nad, afrom), NAD_AVAL(nad, afrom));
222  cur = strstr(from, "@");
223  if(cur != NULL)
224  cur = strstr(cur, "/");
225  else
226  cur = strstr(from, "/");
227  if(cur != NULL) *cur = '\0';
228  }
229 
230  for(acl = r->filter; acl != NULL; acl = acl->next) {
231  if( from == NULL && acl->from != NULL) continue; /* no match if NULL matched vs not-NULL */
232  if( to == NULL && acl->to != NULL ) continue;
233  if( from != NULL && acl->from == NULL) continue; /* no match if not-NULL matched vs NULL */
234  if( to != NULL && acl->to == NULL ) continue;
235  if( from != NULL && acl->from != NULL && fnmatch(acl->from, from, 0) != 0 ) continue; /* do filename-like match */
236  if( to != NULL && acl->to != NULL && fnmatch(acl->to, to, 0) != 0 ) continue;
237  if( acl->what != NULL && nad_find_elem_path(nad, 0, -1, acl->what) < 0 ) continue; /* match packet type */
238  log_debug(ZONE, "matched packet %s->%s vs rule (%s %s->%s)", from, to, acl->what, acl->from, acl->to);
239  if( acl->dump != NULL ) {
240  const char *out;
241  int len;
242  FILE *fd;
243  fd = fopen(acl->dump, "a");
244  if (fd == NULL) {
245  log_write(r->log, LOG_ERR, "filter: cannot open dump file %s: \"%s\", disabling dump for this rule.", acl->dump, strerror(errno));
246  free(acl->dump);
247  acl->dump = NULL;
248  } else {
249  nad_print(nad, 1, &out, &len);
250  fwrite(out, len, 1, fd);
251  /* Add newlines between the stanzas to improve human readability. */
252  fwrite("\n", 1, 1, fd);
253  fclose(fd);
254  }
255  }
256  if (acl->log) {
257  if (acl->redirect) log_write(r->log, LOG_NOTICE, "filter: redirect packet from=%s to=%s - rule (from=%s to=%s what=%s), new to=%s", from, to, acl->from, acl->to, acl->what, acl->redirect);
258  else log_write(r->log, LOG_NOTICE, "filter: %s packet from=%s to=%s - rule (from=%s to=%s what=%s)",(acl->error?"deny":"allow"), from, to, acl->from, acl->to, acl->what);
259  }
260  if (acl->redirect) nad_set_attr(nad, 0, -1, "to", acl->redirect, acl->redirect_len);
261  error = acl->error;
262  break;
263  }
264 
265  if(to != NULL) free(to);
266  if(from != NULL) free(from);
267  return error;
268 }
