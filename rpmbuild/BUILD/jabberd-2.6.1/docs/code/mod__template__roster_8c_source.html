46 static int _template_roster_reload(template_roster_t tr) {
47  struct stat st;
48  FILE *f;
49  long size;
50  char *buf;
51  nad_t nad;
52  int nitems, eitem, ajid, as10n, aname, egroup;
53  item_t item;
54 
55  if(stat(tr->filename, &st) < 0) {
56  log_write(tr->sm->log, LOG_ERR, "couldn't stat roster template %s: %s", tr->filename, strerror(errno));
57  return 1;
58  }
59 
60  if(st.st_mtime <= tr->mtime)
61  return 0;
62 
63  tr->mtime = st.st_mtime;
64 
65  if(tr->items != NULL)
66  xhash_free(tr->items);
67 
68  tr->items = xhash_new(101);
69 
70  f = fopen(tr->filename, "r");
71  if(f == NULL) {
72  log_write(tr->sm->log, LOG_ERR, "couldn't open roster template %s: %s", tr->filename, strerror(errno));
73  return 1;
74  }
75 
76  fseek(f, 0, SEEK_END);
77  size = ftell(f);
78  if(size < 0) {
79  log_write(tr->sm->log, LOG_ERR, "couldn't seek roster template %s: %s", tr->filename, strerror(errno));
80  fclose(f);
81  return 1;
82  }
83  if(size == 0) {
84  log_write(tr->sm->log, LOG_ERR, "empty roster template %s", tr->filename);
85  fclose(f);
86  return 1;
87  }
88  fseek(f, 0, SEEK_SET);
89 
90  buf = (char *) malloc(sizeof(char) * size);
91 
92  if (fread(buf, 1, size, f) != size || ferror(f)) {
93  log_write(tr->sm->log, LOG_ERR, "couldn't read from roster template %s: %s", tr->filename, strerror(errno));
94  free(buf);
95  fclose(f);
96  return 1;
97  }
98 
99  fclose(f);
100 
101  nad = nad_parse(buf, size);
102  if(nad == NULL) {
103  log_write(tr->sm->log, LOG_ERR, "couldn't parse roster template");
104  free(buf);
105  return 1;
106  }
107 
108  free(buf);
109 
110  if(nad->ecur < 2) {
111  log_write(tr->sm->log, LOG_NOTICE, "roster template has no elements");
112  }
113 
114  nitems = 0;
115  eitem = nad_find_elem(nad, 0, NAD_ENS(nad, 0), "item", 1);
116  while(eitem >= 0) {
117  ajid = nad_find_attr(nad, eitem, -1, "jid", NULL);
118  if(ajid < 0) {
119  log_write(tr->sm->log, LOG_ERR, "roster template has item with no jid, skipping");
120  continue;
121  }
122 
123  item = (item_t) pmalloco(xhash_pool(tr->items), sizeof(struct item_st));
124 
125  item->jid = jid_new(NAD_AVAL(nad, ajid), NAD_AVAL_L(nad, ajid));
126  if(item->jid == NULL) {
127  log_write(tr->sm->log, LOG_ERR, "roster template has item with invalid jid, skipping");
128  continue;
129  }
130  pool_cleanup(xhash_pool(tr->items), (void (*)(void *)) jid_free, item->jid);
131 
132  as10n = nad_find_attr(nad, eitem, -1, "subscription", NULL);
133  if(as10n >= 0) {
134  if(NAD_AVAL_L(nad, as10n) == 2 && strncmp("to", NAD_AVAL(nad, as10n), 2) == 0)
135  item->to = 1;
136  else if(NAD_AVAL_L(nad, as10n) == 4 && strncmp("from", NAD_AVAL(nad, as10n), 4) == 0)
137  item->from = 1;
138  else if(NAD_AVAL_L(nad, as10n) == 4 && strncmp("both", NAD_AVAL(nad, as10n), 4) == 0)
139  item->to = item->from = 1;
140  }
141 
142  aname = nad_find_attr(nad, eitem, -1, "name", NULL);
143  if(aname >= 0)
144  item->name = pstrdupx(xhash_pool(tr->items), NAD_AVAL(nad, aname), NAD_AVAL_L(nad, aname));
145 
146  egroup = nad_find_elem(nad, eitem, NAD_ENS(nad, 0), "group", 1);
147  while(egroup >= 0) {
148  if(NAD_CDATA_L(nad, egroup) <= 0) {
149  log_write(tr->sm->log, LOG_ERR, "roster template has zero-length group, skipping");
150  continue;
151  }
152 
153  item->groups = (const char **) realloc(item->groups, sizeof(char *) * (item->ngroups + 1));
154  item->groups[item->ngroups] = pstrdupx(xhash_pool(tr->items), NAD_CDATA(nad, egroup), NAD_CDATA_L(nad, egroup));
155  item->ngroups++;
156 
157  egroup = nad_find_elem(nad, egroup, NAD_ENS(nad, 0), "group", 0);
158  }
159 
160  if(item->groups != NULL)
161  pool_cleanup(xhash_pool(tr->items), free, item->groups);
162 
163  xhash_put(tr->items, jid_full(item->jid), item);
164 
165  log_debug(ZONE, "loaded roster template item %s, %d groups", jid_full(item->jid), item->ngroups);
166 
167  nitems++;
168 
169  eitem = nad_find_elem(nad, eitem, NAD_ENS(nad, 0), "item", 0);
170  }
171 
172  log_write(tr->sm->log, LOG_NOTICE, "loaded %d items from roster template", nitems);
173 
174  return 0;
175 }

178 static void _template_roster_save_item(sm_t sm, jid_t jid, item_t item) {
179  os_t os;
180  os_object_t o;
181  char filter[4096];
182  int i;
183 
184  log_debug(ZONE, "saving roster item %s for %s", jid_full(item->jid), jid_user(jid));
185 
186  os = os_new();
187  o = os_object_new(os);
188 
189  os_object_put(o, "jid", jid_full(item->jid), os_type_STRING);
190 
191  if(item->name != NULL)
192  os_object_put(o, "name", item->name, os_type_STRING);
193 
194  os_object_put(o, "to", &item->to, os_type_BOOLEAN);
195  os_object_put(o, "from", &item->from, os_type_BOOLEAN);
196  os_object_put(o, "ask", &item->ask, os_type_INTEGER);
197 
198  snprintf(filter, 4096, "(jid=%zu:%s)", strlen(jid_full(item->jid)), jid_full(item->jid));
199 
200  storage_replace(sm->st, "roster-items", jid_user(jid), filter, os);
201 
202  os_free(os);
203 
204  snprintf(filter, 4096, "(jid=%zu:%s)", strlen(jid_full(item->jid)), jid_full(item->jid));
205 
206  if(item->ngroups == 0) {
207  storage_delete(sm->st, "roster-groups", jid_user(jid), filter);
208  return;
209  }
210 
211  os = os_new();
212 
213  for(i = 0; i < item->ngroups; i++) {
214  o = os_object_new(os);
215 
216  os_object_put(o, "jid", jid_full(item->jid), os_type_STRING);
217  os_object_put(o, "group", item->groups[i], os_type_STRING);
218  }
219 
220  storage_replace(sm->st, "roster-groups", jid_user(jid), filter, os);
221 
222  os_free(os);
223 }

225 static int _template_roster_user_create(mod_instance_t mi, jid_t jid) {
226  template_roster_t tr = (template_roster_t) mi->mod->private;
227  item_t item;
228  union xhashv xhv;
229 
230  if(_template_roster_reload(tr) != 0)
231  return 0;
232 
233  log_debug(ZONE, "populating roster with items from template");
234 
235  if(xhash_iter_first(tr->items))
236  do {
237  xhv.item_val = &item;
238  xhash_iter_get(tr->items, NULL, NULL, xhv.val);
239 
240  _template_roster_save_item(tr->sm, jid, item);
241  } while(xhash_iter_next(tr->items));
242 
243  return 0;
244 }
