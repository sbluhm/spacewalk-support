23 sx_t sx_new(sx_env_t env, int tag, sx_callback_t cb, void *arg) {
24  sx_t s;
25  int i;
26 
27  assert((int) (cb != NULL));
28 
29  s = (sx_t) calloc(1, sizeof(struct _sx_st));
30 
31  s->env = env;
32  s->tag = tag;
33  s->cb = cb;
34  s->cb_arg = arg;
35 
36  s->expat = XML_ParserCreateNS(NULL, '|');
37  XML_SetReturnNSTriplet(s->expat, 1);
38  XML_SetUserData(s->expat, (void *) s);
39  /* Prevent the "billion laughs" attack against expat by disabling
40  * internal entity expansion. With 2.x, forcibly stop the parser
41  * if an entity is declared - this is safer and a more obvious
42  * failure mode. With older versions, simply prevent expenansion
43  * of such entities. */
44 #ifdef HAVE_XML_STOPPARSER
45  XML_SetEntityDeclHandler(s->expat, (void *) _sx_entity_declaration);
46 #else
47  XML_SetDefaultHandler(s->expat, NULL);
48 #endif
49 
50 #ifdef HAVE_XML_SETHASHSALT
51  XML_SetHashSalt(s->expat, rand());
52 #endif
53 
54  s->wbufq = jqueue_new();
55  s->rnadq = jqueue_new();
56 
57  if(env != NULL) {
58  s->plugin_data = (void **) calloc(1, sizeof(void *) * env->nplugins);
59 
60  for(i = 0; i < env->nplugins; i++)
61  if(env->plugins[i]->new != NULL)
62  (env->plugins[i]->new)(s, env->plugins[i]);
63  }
64 
65  _sx_debug(ZONE, "allocated new sx for %d", tag);
66 
67  return s;
68 }

100

110 
111  if(s->auth_method != NULL) free((void*)s->auth_method);

116  for(i = 0; i < s->env->nplugins; i++)
117  if(s->env->plugins[i]->free != NULL)
118  (s->env->plugins[i]->free)(s, s->env->plugins[i]);

154 void _sx_reset(sx_t s) {
155  struct _sx_st temp;
156  sx_t new;
157 
158  _sx_debug(ZONE, "resetting stream state");
159 
160  /* we want to reset the contents of s, but we can't free s because
161  * the caller (and others) hold references. so, we make a new sx_t,
162  * copy the contents (only pointers), free it (which will free strings
163  * and queues), then make another new one, and copy the contents back
164  * into s */
165 
166  temp.env = s->env;
167  temp.tag = s->tag;
168  temp.cb = s->cb;
169  temp.cb_arg = s->cb_arg;
170 
171  temp.ip = s->ip;
172  temp.port = s->port;
173  temp.flags = s->flags;
174  temp.reentry = s->reentry;
175  temp.ssf = s->ssf;
176  temp.wio = s->wio;
177  temp.rio = s->rio;
178  temp.wnad = s->wnad;
179  temp.rnad = s->rnad;
180  temp.rbytesmax = s->rbytesmax;
181  temp.plugin_data = s->plugin_data;
182 
183  s->reentry = 0;
184 
185  s->env = NULL; /* we get rid of this, because we don't want plugin data to be freed */
186 
187  new = (sx_t) malloc(sizeof(struct _sx_st));
188  memcpy(new, s, sizeof(struct _sx_st));
189  sx_free(new);
190 
191  new = sx_new(NULL, temp.tag, temp.cb, temp.cb_arg);
192  memcpy(s, new, sizeof(struct _sx_st));
193  free(new);
194 
195  /* massaged expat into shape */
196  XML_SetUserData(s->expat, (void *) s);
197 
198  s->env = temp.env;
199  s->ip = temp.ip;
200  s->port = temp.port;
201  s->flags = temp.flags;
202  s->reentry = temp.reentry;
203  s->ssf = temp.ssf;
204  s->wio = temp.wio;
205  s->rio = temp.rio;
206  s->wnad = temp.wnad;
207  s->rnad = temp.rnad;
208  s->rbytesmax = temp.rbytesmax;
209  s->plugin_data = temp.plugin_data;
210 
211  s->has_reset = 1;
212 
213  _sx_debug(ZONE, "finished resetting stream state");
214 }

262 void _sx_buffer_alloc_margin(sx_buf_t buf, int before, int after)
263 {
264  char *new_heap;
265 
266  assert( before >= 0 );
267  assert( after >= 0 );
268 
269  /* If there wasn't any data in the buf, we can just allocate space for the margins */
270  if (buf->data == NULL || buf->len == 0) {
271  if (buf->heap != NULL)
272  buf->heap = realloc(buf->heap, before+after);
273  else
274  buf->heap = malloc(before+after);
275  buf->data = buf->heap + before;
276  return;
277  }
278 
279  if (buf->heap != NULL) {
280  int old_leader = buf->data - buf->heap;
281  /* Hmmm, maybe we can just call realloc() ? */
282  if (old_leader >= before && old_leader <= (before * 4)) {
283  buf->heap = realloc(buf->heap, before + buf->len + after);
284  buf->data = buf->heap + old_leader;
285  return;
286  }
287  }
288 
289  /* Most general case --- allocate a new buffer, copy stuff over, free the old one. */
290  new_heap = malloc(before + buf->len + after);
291  memcpy(new_heap + before, buf->data, buf->len);
292  if (buf->heap != NULL)
293  free(buf->heap);
294  buf->heap = new_heap;
295  buf->data = new_heap + before;
296 }
